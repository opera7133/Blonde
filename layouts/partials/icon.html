{{- /* Usage: {{ partial "icon.html" "menu" }} {{ partial "icon.html" (dict "name" "schedule" "class" "mr-2") }} Parameters: name (string) — filename without .svg under assets/icons/ class (string) — extra CSS classes appended to the wrapping The SVG fill is replaced with currentColor so the icon inherits the surrounding text color. Width/height attributes are removed and replaced with 1em so the icon scales with font-size. */ -}} {{- $name := . -}} {{- $extra := "" -}} {{- if reflect.IsMap . -}} {{- $name = index . "name" -}} {{- $extra = index . "class" | default "" -}} {{- end -}} {{- $icon := resources.Get (printf "icons/%s.svg" $name) -}} {{- with $icon -}} {{- $c := .Content -}} {{- $c = $c | replaceRE ` (width|height)="[^"]*"` ` $1="1em"` -}} {{- $c = $c | replaceRE `fill="[^"]*"` `fill="currentColor"` -}} {{ $c | safeHTML }} {{- end -}}
